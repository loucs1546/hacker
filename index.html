  <script>

    // ======================

    // GÉNÉRATION DU FOND BINAIRE

    // ======================

    const binaryBg = document.getElementById('binary-bg');

    const chars = '01';

    const lines = 50;


    for (let i = 0; i < lines; i++) {

      const line = document.createElement('div');

      line.className = 'binary-line';

      line.style.top = `${Math.random() * 100}%`;

      line.style.left = `${Math.random() * 100}%`;

      line.style.animationDuration = `${Math.random() * 10 + 5}s`;

      line.style.animationDelay = `${Math.random() * 5}s`;


      let text = '';

      for (let j = 0; j < 80; j++) {

        text += chars[Math.floor(Math.random() * 2)];

      }

      line.textContent = text;


      binaryBg.appendChild(line);

    }


    // ======================

    // ANIMATION DE LA BARRE ET DU TEXTE

    // ======================

    const statusEl = document.getElementById('status');

    const progressEl = document.getElementById('progress');

    const completeEl = document.getElementById('complete');


    let progress = 0;

    const interval = setInterval(() => {

      if (progress < 100) {

        progress += Math.random() * 3 + 0.5;

        if (progress > 100) progress = 100;


        progressEl.style.width = `${progress}%`;


        if (progress < 30) {

          statusEl.textContent = "INITIALISATION DES DONNÉES...";

        } else if (progress < 70) {

          statusEl.textContent = "CHARGEMENT EN COURS...";

        } else {

          statusEl.textContent = "FINALISATION...";

        }

      } else {

        clearInterval(interval);

        statusEl.textContent = "TERMINÉ";

        statusEl.classList.add('glow');

        completeEl.style.opacity = 1;

        setTimeout(() => {

          completeEl.classList.add('glow');

        }, 500);

      }

    }, 150);


    // ======================

    // EFFET DE PULSATION SUR LA FENÊTRE

    // ======================

    document.querySelector('.hacking-window').addEventListener('mouseenter', () => {

      document.querySelector('.hacking-window').style.boxShadow = '0 0 40px #00ff41';

    });


    document.querySelector('.hacking-window').addEventListener('mouseleave', () => {

      document.querySelector('.hacking-window').style.boxShadow = '0 0 20px #00ff41';

    });
